201  template <typename T>
202  void UniversalTransfer(std::span<T> buffer)
203  {
204  // Follows the instructions provided by RM0008.pdf page 710 to get the
205  // maximum performance.
206 
207  // Step 1. Load the first byte into the data register
208  port_.spi->DR = buffer[0];
209 
210  for (size_t i = 1; i < buffer.size(); i++)
211  {
212  // Step 2.1 Wait until the transmit buffer is empty, then...
213  while (!TransmitBufferEmpty())
214  {
215  continue;
216  }
217 
218  // Step 2.2 Load it with the next byte.
219  port_.spi->DR = buffer[i];
220 
221  // Step 2.3 Wait until the receive buffer has data.
222  while (!ReceiveBufferNotEmpty())
223  {
224  continue;
225  }
226 
227  // Step 2.4 Read byte from DR register and place it into the previous byte
228  // location. This is important as the bytes written are head of the bytes
229  // received by 1.
230  buffer[i - 1] = static_cast<T>(port_.spi->DR);
231  }
232 
233  // Step 3.1 Wait for the last byte to come in on the receive buffer.
234  while (!ReceiveBufferNotEmpty())
235  {
236  continue;
237  }
238 
239  // Step 3.2 Read the final byte from the receive buffer and place it in the
240  // end of the span.
241  buffer.end()[-1] = static_cast<T>(port_.spi->DR);
242  }
