42  (adc_.Voltage() / kPullDownResistance);
43  // From Fig. 3 - Photo Current vs. Illuminance of TEMT6000X01 datasheet,
44  // Solve for x using log(y) = m * log(x) + b, where x is the
45  // illuminance in lux and y is the current in micro-amps.
46  //
47  // Plugging in (20, 10) and (100, 50) to solve for m:
48  // m = ∆log(y) / ∆log(x)
49  // = log(y2 / y1) / log(x2 / x1)
50  // = log(50 / 10) / log(100 / 20)
51  // = log(5) / log(5)
52  // m = 1
53  //
54  // Plugging in (100, 50) to solve for b:
55  // log(y) = log(x) + b
56  // log(50) = log(100) + b
57  // b = -log(2)
58  //
59  // Solve for x:
60  // log(y) = log(x) - log(2)
61  // = log(x / 2)
62  // --> y = x / 2
63  // --> x = 2 * y
64  return units::illuminance::lux_t(2.0f * kCurrent.to<float>());
