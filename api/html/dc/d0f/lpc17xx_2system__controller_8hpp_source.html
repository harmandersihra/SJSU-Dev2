472  void Initialize() override
473  {
474  units::frequency::hertz_t osc_clk =
475  clock_configuration_.main_oscillator.frequency;
476  units::frequency::hertz_t sys_clk = 0_MHz;
477  units::frequency::hertz_t pll_clk = 0_MHz;
478  units::frequency::hertz_t usb_clk = 0_MHz;
479  uint8_t cpu_clock_divider = clock_configuration_.cpu.divider;
480 
481  // =========================================================================
482  // Step 1. Set the internal RC as the clock source for all clocks.
483  // =========================================================================
484  // Disabling and disconnecting the PLLs sets the CPU and USB clock source to
485  // be the system clock.
486  DisableAndDisconnectPll(&system_controller->PLL0CON,
487  &system_controller->PLL0FEED);
488  DisableAndDisconnectPll(&system_controller->PLL1CON,
489  &system_controller->PLL1FEED);
490  // Disable the main oscillator.
491  SystemControlsRegister::Register()
492  .Clear(SystemControlsRegister::kOscillatorEnableMask)
493  .Save();
494  SetSystemClockSource(Oscillator::kIrc);
495 
496  // =========================================================================
497  // Step 2. Enable the main oscillator if it is used to drive the system
498  // clock and/or PLL1.
499  // =========================================================================
500  if ((clock_configuration_.usb.clock_source == UsbClockSource::kPll1) ||
501  (clock_configuration_.system.clock_source == Oscillator::kMain))
502  {
503  EnableMainOscillator();
504  }
505 
506  // =========================================================================
507  // Step 3. Determine the running system clock rate based on the selected
508  // oscillator.
509  // =========================================================================
510  switch (clock_configuration_.system.clock_source)
511  {
512  case Oscillator::kIrc: sys_clk = kIrcFrequency; break;
513  case Oscillator::kMain: sys_clk = osc_clk; break;
514  case Oscillator::kRtc: sys_clk = kRTCFrequency; break;
515  }
516 
517  // =========================================================================
518  // Step 4. Configure the CPU clock by configuring and enabling PLL0 if
519  // necessary.
520  // =========================================================================
521  switch (clock_configuration_.cpu.clock_source)
522  {
523  case SystemController::CpuClockSource::kSystemClock:
524  {
525  SetSystemClockSource(clock_configuration_.system.clock_source);
526  SetCpuClockDivider(cpu_clock_divider);
527  pll_clk = sys_clk;
528  break;
529  }
530  case SystemController::CpuClockSource::kPll0:
531  {
532  // Before configuring PLL0, verify the PLL0 configurations by checking
533  // whether the expected PLL0 current controlled oscillator frequency and
534  // the expected CPU frequency are within the valid frequency ranges.
535 
536  // Minimum / maximum current controlled oscillator frequencies of PLL0.
537  constexpr units::frequency::hertz_t kMinFcco = 275_MHz;
538  constexpr units::frequency::hertz_t kMaxFcco = 550_MHz;
539 
540  uint16_t multiplier = clock_configuration_.pll0.multiplier;
541  uint8_t pre_divider = clock_configuration_.pll0.pre_divider;
542  // Determine the running current controlled oscillator frequency based
543  // on the configurations.
544  pll_clk = (2 * sys_clk * multiplier) / pre_divider;
545 
546  SJ2_ASSERT_FATAL(
547  (kMinFcco < pll_clk) && (pll_clk < kMaxFcco),
548  "Invalid PLL0 multiplier and pre-divider configurations. The "
549  "target current controlled oscillator frequency must be between "
550  "275 MHz and 550 MHz.");
551 
552  auto max_cpu_clock_rate = clock_configuration_.cpu.max_cpu_clock_rate;
553 
554  SJ2_ASSERT_FATAL((pll_clk / cpu_clock_divider) < max_cpu_clock_rate,
555  "The configured CPU divider produces a CPU clock that "
556  "exceeds the maximum allowed CPU clock rate.");
557 
558  ConfigurePll0();
559  break;
560  }
561  }
562 
563  // =========================================================================
564  // Step 5. Configure the USB clock by configuring and enabling PLL1 if
565  // necessary.
566  // =========================================================================
567  switch (clock_configuration_.usb.clock_source)
568  {
569  case UsbClockSource::kPll0:
570  {
571  uint8_t usb_clock_divider_select =
572  Value(clock_configuration_.usb.divider);
573  uint32_t usb_clock_divider = usb_clock_divider_select + 1;
574  usb_clk = pll_clk / usb_clock_divider;
575  UsbClockRegister::Register()
576  .Insert(usb_clock_divider_select, UsbClockRegister::kDividerMask)
577  .Save();
578  break;
579  }
580  case UsbClockSource::kPll1:
581  {
582  uint32_t multiplier = Value(clock_configuration_.pll1.multiplier) + 1;
583  usb_clk = osc_clk * multiplier;
584 
585  ConfigurePll1();
586  break;
587  }
588  }
589 
590  // =========================================================================
591  // Step 6. Set all peripheral clock dividers to 1.
592  // =========================================================================
593  // SEE: 4.7.3 Peripheral Clock Selection registers 0 and 1
594  // https://www.nxp.com/docs/en/user-guide/UM10360.pdf#page=58
595  //
596  // NOTE: The reserved bits are left as 0b00.
597  system_controller->PCLKSEL0 = 0x5551'5155;
598  system_controller->PCLKSEL0 = 0x5455'5455;
599 
600  // =========================================================================
601  // Step 7. Determine the running CPU and USB clock rates.
602  // =========================================================================
603  cpu_clock_rate_ = pll_clk / cpu_clock_divider;
604  usb_clock_rate_ = usb_clk;
605  }

784  void ConfigurePll0() const
785  {
786  volatile uint32_t * pll_feed_register = &system_controller->PLL0FEED;
787  volatile uint32_t * pll_status_register = &system_controller->PLL0STAT;
788 
789  // The following sequence is specified in the LPC176x/5x User Manual
790  // Section 4.5.13 and must be followed step by step in order to have PLL0
791  // initialized and running.
792 
793  // NOTE: It is very important not to merge any steps. For example, do not
794  // update the PLL0CFG and enable PLL0 simultaneously with the same
795  // feed sequence.
796 
797  // =========================================================================
798  // Step 1. Disconnect PLL0 with one feed sequence if PLL0 is already
799  // connected.
800  //
801  // Step 2. Disable PLL0 with one feed sequence.
802  // =========================================================================
803  // NOTE: DisableAndDisconnectPll performs Steps 1 and 2.
804  DisableAndDisconnectPll(&system_controller->PLL0CON, pll_feed_register);
805 
806  // =========================================================================
807  // Step 3. Change the CPU Clock Divider setting to speed up operation
808  // without PLL0, if desired.
809  // =========================================================================
810  SetCpuClockDivider(1);
811 
812  // =========================================================================
813  // Step 4. Write to the Clock Source Selection Control register to change
814  // the clock source if needed.
815  // =========================================================================
816  SetSystemClockSource(clock_configuration_.system.clock_source);
817 
818  // =========================================================================
819  // Step 5. Write to the PLL0CFG and make it effective with one feed
820  // sequence. The PLL0CFG can only be updated when PLL0 is disabled.
821  // =========================================================================
822  uint16_t multiplier = clock_configuration_.pll0.multiplier;
823  uint8_t pre_divider = clock_configuration_.pll0.pre_divider;
824  Pll0::ConfigurationRegister::Register()
825  .Insert((multiplier - 1), Pll0::ConfigurationRegister::kMultiplierMask)
826  .Insert((pre_divider - 1), Pll0::ConfigurationRegister::kPreDividerMask)
827  .Save();
828  WritePllFeedSequence(pll_feed_register);
829 
830  // =========================================================================
831  // Step 6. Enable PLL0 with one feed sequence.
832  // =========================================================================
833  PllControlRegister::Register()[0]
834  .Set(PllControlRegister::kEnableMask)
835  .Save();
836  WritePllFeedSequence(pll_feed_register);
837 
838  // =========================================================================
839  // Step 7. Change the CPU Clock Divider setting for the operation with PLL0.
840  // It is critical to do this before connecting PLL0.
841  // =========================================================================
842  SetCpuClockDivider(clock_configuration_.cpu.divider);
843 
844  // =========================================================================
845  // Step 8. Wait for PLL0 to achieve lock by monitoring the PLOCK0 bit in the
846  // PLL0STAT register, or using the PLOCK0 interrupt, or wait for a
847  // fixed time when the input clock to PLL0 is slow (i.e. 32 kHz).
848  // =========================================================================
849  WaitForPllLockStatus(pll_status_register,
850  Pll0::StatusRegister::kLockStatusMask);
851 
852  // =========================================================================
853  // Step 9. Connect PLL0 with one feed sequence.
854  // =========================================================================
855  PllControlRegister::Register()[0]
856  .Set(PllControlRegister::kConnectMask)
857  .Save();
858  WritePllFeedSequence(pll_feed_register);
859 
860  WaitForPllConnectionStatus(pll_status_register,
861  Pll0::StatusRegister::kEnableMask,
862  Pll0::StatusRegister::kConnectMask);
863  }
