118  Acceleration_t Read() override
119  {
120  Acceleration_t acceleration = {};
121  std::array<int16_t, 3> result = memory_[Map::kXYZStartAddress];
122  auto xyz_data = result;
123 
124  // First X-axis Byte (MSB first)
125  // =========================================================================
126  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
127  // XD11 | XD10 | XD9 | XD8 | XD7 | XD6 | XD5 | XD4
128  //
129  // Final X-axis Byte (LSB)
130  // =========================================================================
131  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
132  // XD3 | XD2 | XD1 | XD0 | 0 | 0 | 0 | 0
133  //
134  // We simply shift and OR the bytes together to get them into a signed int
135  // 16 value. We do not shift yet because we want to get the signed bit in
136  // the most significant bit position to allow for sign extension when we
137  // shift to the right later.
138 
139  int16_t x = static_cast<int16_t>(xyz_data[0] >> 4);
140  int16_t y = static_cast<int16_t>(xyz_data[1] >> 4);
141  int16_t z = static_cast<int16_t>(xyz_data[2] >> 4);
142 
143  // Convert the 12-bit signed value into a value from -1.0 to 1.0f so it can
144  // be multiplied by the full_scale in order to get the true acceleration.
145  constexpr int16_t kMin = sjsu::BitLimits<12, int16_t>::Min();
146  constexpr int16_t kMax = sjsu::BitLimits<12, int16_t>::Max();
147 
148  float x_axis_ratio = sjsu::Map(x, kMin, kMax, -1.0f, 1.0f);
149  float y_axis_ratio = sjsu::Map(y, kMin, kMax, -1.0f, 1.0f);
150  float z_axis_ratio = sjsu::Map(z, kMin, kMax, -1.0f, 1.0f);
151 
152  acceleration.x = full_scale_ * x_axis_ratio;
153  acceleration.y = full_scale_ * y_axis_ratio;
154  acceleration.z = full_scale_ * z_axis_ratio;
155
