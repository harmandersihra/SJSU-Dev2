88  // First X-axis Byte (MSB first)
89  // =========================================================================
90  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
91  // XD15 | XD14 | XD13 | XD12 | XD11 | XD10 | XD9 | XD8
92  //
93  // Final X-axis Byte (LSB)
94  // =========================================================================
95  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
96  // XD7 | XD6 | XD5 | XD4 | XD3 | XD2 | XD1 | XD0
97  //
98  // We simply shift and OR the bytes together to get them into a signed int
99  // 16 value.
100 
101  int16_t x = static_cast<int16_t>(xyz_data[0] << 8 | xyz_data[1]);
102  int16_t y = static_cast<int16_t>(xyz_data[2] << 8 | xyz_data[3]);
103  int16_t z = static_cast<int16_t>(xyz_data[4] << 8 | xyz_data[5]);
104 
105  // Convert the 16 bit value into a floating point value m/S^2
106  constexpr int16_t kMax = std::numeric_limits<int16_t>::max();
107  constexpr int16_t kMin = std::numeric_limits<int16_t>::min();
108 
109  float x_ratio = sjsu::Map(x, kMin, kMax, -1.0f, 1.0f);
110  float y_ratio = sjsu::Map(y, kMin, kMax, -1.0f, 1.0f);
111  float z_ratio = sjsu::Map(z, kMin, kMax, -1.0f, 1.0f);
112
