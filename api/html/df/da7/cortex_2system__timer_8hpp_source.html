75  static std::chrono::nanoseconds GetCount()
76  {
77  // Capture all count variables at this instant.
78  uint32_t high_speed_ticks = dwt_counter.GetCount();
79  auto current_millis_count = millisecond_count;
80 
81  // Get rid of the portion of the dwt tick count that is above 1ms.
82  // We only want the information about the ticks that is below 1ms, since we
83  // get the 1ms count from the SysTick timer interrupt.
84  uint32_t sub_millis_ticks = (high_speed_ticks % ticks_per_millisecond);
85 
86  // Perform the conversion from ticks to nanoseconds
87  auto nanoseconds_unscaled = (sub_millis_ticks * nanoseconds_per_tick);
88 
89  // Nanoseconds per tick is actually scaled up by the kFixedPointScaling
90  // factor, meaning it needs to be divided by that value to get the correct
91  // nanosecond_uptime.
92  auto nanoseconds_uptime_scaled = nanoseconds_unscaled / kFixedPointScaling;
93 
94  // Result is simply the uptime milliseconds (represented in nanoseconds) +
95  // the added nanoseconds
96  return current_millis_count + nanoseconds_uptime_scaled;
97  }
