141  static void I2cHandler(const Bus_t & i2c)
142  {
143  MasterState state = MasterState(i2c.registers->STAT);
144  uint32_t clear_mask = 0;
145  uint32_t set_mask = 0;
146  switch (state)
147  {
148  case MasterState::kBusError: // 0x00
149  {
150  i2c.transaction.status = std::errc::io_error;
151  set_mask = Control::kAssertAcknowledge | Control::kStop;
152  break;
153  }
154  case MasterState::kStartCondition: // 0x08
155  {
156  i2c.registers->DAT = i2c.transaction.GetProperAddress();
157  break;
158  }
159  case MasterState::kRepeatedStart: // 0x10
160  {
161  i2c.transaction.operation = Operation::kRead;
162  i2c.registers->DAT = i2c.transaction.GetProperAddress();
163  break;
164  }
165  case MasterState::kSlaveAddressWriteSentReceivedAck: // 0x18
166  {
167  clear_mask = Control::kStart;
168  if (i2c.transaction.out_length == 0)
169  {
170  i2c.transaction.busy = false;
171  set_mask = Control::kStop;
172  }
173  else
174  {
175  size_t position = i2c.transaction.position++;
176  i2c.registers->DAT = i2c.transaction.data_out[position];
177  }
178  break;
179  }
180  case MasterState::kSlaveAddressWriteSentReceivedNack: // 0x20
181  {
182  clear_mask = Control::kStart;
183  i2c.transaction.busy = false;
184  i2c.transaction.status = std::errc::no_such_device_or_address;
185  set_mask = Control::kStop;
186  break;
187  }
188  case MasterState::kTransmittedDataReceivedAck: // 0x28
189  {
190  if (i2c.transaction.position >= i2c.transaction.out_length)
191  {
192  if (i2c.transaction.repeated)
193  {
194  // OR with 1 to set address as READ for the next transaction
195  i2c.transaction.operation = Operation::kRead;
196  i2c.transaction.position = 0;
197  set_mask = Control::kStart;
198  }
199  else
200  {
201  i2c.transaction.busy = false;
202  set_mask = Control::kStop;
203  }
204  }
205  else
206  {
207  size_t position = i2c.transaction.position++;
208  i2c.registers->DAT = i2c.transaction.data_out[position];
209  }
210  break;
211  }
212  case MasterState::kTransmittedDataReceivedNack: // 0x30
213  {
214  i2c.transaction.busy = false;
215  set_mask = Control::kStop;
216  break;
217  }
218  case MasterState::kArbitrationLost: // 0x38
219  {
220  set_mask = Control::kStart;
221  break;
222  }
223  case MasterState::kSlaveAddressReadSentReceivedAck: // 0x40
224  {
225  clear_mask = Control::kStart;
226  if (i2c.transaction.in_length == 0)
227  {
228  set_mask = Control::kStop;
229  }
230  // If we only want 1 byte, make sure to nack that byte
231  else if (i2c.transaction.in_length == 1)
232  {
233  clear_mask |= Control::kAssertAcknowledge;
234  }
235  // If we want more then 1 byte, make sure to ack the first byte
236  else
237  {
238  set_mask = Control::kAssertAcknowledge;
239  }
240  break;
241  }
242  case MasterState::kSlaveAddressReadSentReceivedNack: // 0x48
243  {
244  clear_mask = Control::kStart;
245  i2c.transaction.status = std::errc::no_such_device_or_address;
246  i2c.transaction.busy = false;
247  set_mask = Control::kStop;
248  break;
249  }
250  case MasterState::kReceivedDataReceivedAck: // 0x50
251  {
252  const size_t kBufferEnd = i2c.transaction.in_length;
253  if (i2c.transaction.position < kBufferEnd)
254  {
255  const size_t kPosition = i2c.transaction.position;
256  i2c.transaction.data_in[kPosition] =
257  static_cast<uint8_t>(i2c.registers->DAT);
258  i2c.transaction.position++;
259  }
260  // Check if the position has been pushed past the buffer length
261  if (i2c.transaction.position + 1 >= kBufferEnd)
262  {
263  clear_mask = Control::kAssertAcknowledge;
264  i2c.transaction.busy = false;
265  }
266  else
267  {
268  set_mask = Control::kAssertAcknowledge;
269  }
270  break;
271  }
272  case MasterState::kReceivedDataReceivedNack: // 0x58
273  {
274  i2c.transaction.busy = false;
275  if (i2c.transaction.in_length != 0)
276  {
277  size_t position = i2c.transaction.position++;
278  i2c.transaction.data_in[position] =
279  static_cast<uint8_t>(i2c.registers->DAT);
280  }
281  set_mask = Control::kStop;
282  break;
283  }
284  case MasterState::kDoNothing: // 0xF8
285  {
286  break;
287  }
288  default:
289  {
290  clear_mask = Control::kStop;
291  SJ2_ASSERT_FATAL(false, "Invalid I2C State Reached!!");
292  break;
293  }
294  }
295  // Clear I2C Interrupt flag
296  clear_mask |= Control::kInterrupt;
297  // Set register controls
298  i2c.registers->CONSET = set_mask;
299  i2c.registers->CONCLR = clear_mask;
300  }
