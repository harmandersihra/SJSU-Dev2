191  void ModuleInitialize() override
192  {
193  if (!(1 <= channel_.channel && channel_.channel <= 6))
194  {
195  throw Exception(std::errc::invalid_argument,
196  "Channel must be between 1 and 6 on LPC40xx platforms.");
197  }
198  auto & system = sjsu::SystemController::GetPlatformController();
199  system.PowerUpPeripheral(channel_.peripheral.id);
200 
201  // Get a pointer to the PWM registers
202  auto * pwm = channel_.peripheral.registers;
203 
204  // Set prescalar to 1 so the input frequency to the PWM peripheral is
205  // equal to the peripheral clock frequency.
206  //
207  // PR = Prescale register = defines the maximum value for the prescaler
208  pwm->PR = 0;
209 
210  // PC = Prescale counter. Set to 0 to cause the timer counter to increment
211  // after each peripheral clock tick.
212  pwm->PC = 0;
213 
214  // Mode 0x0 means increment time counter (TC) after the peripheral clock
215  // has cycled the amount inside of the prescale.
216  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kMode);
217 
218  // 0x0 for this register says to use the TC for input counts.
219  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kCountInput);
220 
221  // Sets match register 0 to reset when TC and Match 0 match each other,
222  // meaning that the PWM pulse will cycle continuously.
223  pwm->MCR = bit::Set(pwm->MCR, MatchControl::kPwm0Reset);
224 
225  // Enables PWM[channel] output
226  const uint32_t kOutputEnableStart = OutputControl::kEnableOutput.position;
227  const uint32_t kChannelBitPosition = kOutputEnableStart + channel_.channel;
228  pwm->PCR = bit::Set(pwm->PCR, kChannelBitPosition);
229 
230  // Set the pin function
231  channel_.pin.ConfigureFunction(channel_.pin_function_code);
232  }

235  {

237 
238  if (enable)

245

259  void ConfigureFrequency(units::frequency::hertz_t frequency_hz) override
260  {
261  if (frequency_hz <= 0_Hz)
262  {

269 
270  ModuleEnable(false);
271 
272  auto & system = sjsu::SystemController::GetPlatformController();
273  auto peripheral_frequency = system.GetClockRate(channel_.peripheral.id);
274 
275  // Get the current duty cycle so we can match it to the updated frequency.
276  float previous_duty_cycle = GetDutyCycle();
277 
278  // In order to avoid PWM glitches, the PWM must be disabled while updating
279  // the MR0 register. Doing this will reset all counters to 0 and allow us to
280  // update MR0.
281  const auto kNewFrequency = peripheral_frequency / frequency_hz;
282  GetMatchRegisters(0) = kNewFrequency.to<uint32_t>();
283 
284  // Re-enable PWM which will also reset all of the counters which allow
285  // setting the duty cycle of other PWM channels.
286  ModuleEnable(true);
287 
288  SetDutyCycle(previous_duty_cycle);

291  void SetDutyCycle(float duty_cycle) override
292  {
293  // Clamp duty cycle
294  const float kClampedDuty = std::clamp(duty_cycle, 0.0f, 1.0f);
295 
296  // Set match register for this channel
297  GetMatchRegisters(channel_.channel) = CalculateDutyCycle(kClampedDuty);
298 
299  // LER (Load Enable Register) sets the hardware up to load the new duty
300  // cycle period in the next reset cycle after MR0 match. Hardware requires
301  // this as a means ot prevent itself from glitching and changing pwm
302  // instantaniously.
303  channel_.peripheral.registers->LER |= (1 << channel_.channel);
304  }
305 
306  float GetDutyCycle() override
307  {
308  auto period = static_cast<float>(GetMatchRegisters(channel_.channel));
309  auto max_period = static_cast<float>(GetMatchRegisters(0));
310  return period / max_period;
311  }
