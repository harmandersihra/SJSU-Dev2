119 constexpr static UartCalibration_t GenerateUartCalibration(
120  uint32_t baud_rate,
121  units::frequency::hertz_t peripheral_frequency)
122 {
123  uint32_t integer_peripheral_frequency =
124  units::unit_cast<uint32_t>(peripheral_frequency);
125  States state = States::kCalculateIntegerDivideLatch;
126  UartCalibration_t uart_calibration;
127  float baud_rate_float = static_cast<float>(baud_rate);
128  float divide_estimate = 0;
129  float decimal = 1.5;
130  float div = 1;
131  float mul = 2;
132  while (state != States::kDone)
133  {
134  switch (state)
135  {
136  case States::kCalculateIntegerDivideLatch:
137  {
138  divide_estimate =
139  DividerEstimate(baud_rate_float, 1, integer_peripheral_frequency);
140 
141  if (divide_estimate < 1.0f)
142  {
143  uart_calibration.divide_latch = 0;
144  state = States::kDone;
145  }
146  else if (IsDecimal(divide_estimate))
147  {
148  uart_calibration.divide_latch =
149  static_cast<uint32_t>(divide_estimate);
150  state = States::kDone;
151  }
152  else
153  {
154  state = States::kCalculateDivideLatchWithDecimal;
155  }
156  break;
157  }
158  case States::kCalculateDivideLatchWithDecimal:
159  {
160  divide_estimate = RoundFloat(DividerEstimate(
161  baud_rate_float, decimal, integer_peripheral_frequency));
162  decimal = FractionalEstimate(baud_rate_float, divide_estimate,
163  integer_peripheral_frequency);
164  if (1.1f <= decimal && decimal <= 1.9f)
165  {
166  state = States::kGenerateFractionFromDecimal;
167  }
168  else
169  {
170  state = States::kDecimalFailedGenerateNewDecimal;
171  }
172  break;
173  }
174  case States::kDecimalFailedGenerateNewDecimal:
175  {
176  mul += 1;
177 
178  if (div > 15)
179  {
180  state = States::kDone;
181  break;
182  }
183  else if (mul > 15)
184  {
185  div += 1;
186  mul = div + 1;
187  }
188  decimal = div / mul;
189  state = States::kCalculateDivideLatchWithDecimal;
190  break;
191  }
192  case States::kGenerateFractionFromDecimal:
193  {
194  uart_calibration = FindClosestFractional(decimal);
195  uart_calibration.divide_latch = static_cast<uint32_t>(divide_estimate);
196  state = States::kDone;
197  break;
198  }
199  case States::kDone:
200  default: break;
201  }
202  }
203  return uart_calibration;
204 }
