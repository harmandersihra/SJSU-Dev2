103 template <typename T>
104 [[nodiscard]] constexpr T Extract(T target,
105  uint32_t position,
106  uint32_t width = 1)
107 {
108  // Check the types at compile time
109  static_assert(std::numeric_limits<T>::is_integer,
110  "Extract only accepts intergers.");
111  // Need to use an unsigned version of the type T for the mask to make sure
112  // that the shift right doesn't result in a sign extended shift.
113  using UnsignedT = typename std::make_unsigned<T>::type;
114  // At compile time, generate variable containing all 1s with the size of the
115  // target parameter.
116  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
117  // At compile time calculate the number of bits in the target parameter.
118  constexpr size_t kTargetWidth = sizeof(T) * 8;
119  // Create mask by shifting the set of 1s down so that the number of 1s from
120  // bit position 0 is equal to the width parameter.
121  UnsignedT mask =
122  static_cast<UnsignedT>(kFieldOfOnes >> (kTargetWidth - width));
123  // Shift target down to the right to get the bit position in the 0th bit
124  // location.
125  // Mask the value to clear any bit after the width's amount of bits.
126  return static_cast<T>((target >> position) & mask);
127 }

182  uint32_t width = 1)
183 {
184  // Check the types at compile time
185  static_assert(
186  std::numeric_limits<T>::is_integer,
187  "1st parameter (target) of Insert function must be an interger.");
188  static_assert(
189  std::numeric_limits<U>::is_integer,
190  "2nd parameter (value) of Insert function must be an interger.");
191  static_assert(sizeof(T) >= sizeof(U),
192  "2nd parameter (value) must be smaller than or equal to the "
193  "size of the target value.");
194  // Need to use an unsigned version of the type T for the mask to make sure
195  // that the shift right doesn't result in a sign extended shift.
196  using UnsignedT = typename std::make_unsigned<T>::type;
197  // At compile time, generate variable containing all 1s with the size of the
198  // target parameter.
199  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
200  // At compile time calculate the number of bits in the target parameter.
201  constexpr size_t kTargetWidth = sizeof(T) * 8;
202  // Create mask by shifting the set of 1s down so that the number of 1s from
203  // bit position 0 is equal to the width parameter.
204  UnsignedT mask =
205  static_cast<UnsignedT>(kFieldOfOnes >> (kTargetWidth - width));
206  // Clear width's number of bits in the target value at the bit position
207  // specified.
208  target = static_cast<UnsignedT>(target & ~(mask << position));
209  // AND value with mask to remove any bits beyond the specified width.
210  // Shift masked value into bit position and OR with target value.
211  target = static_cast<UnsignedT>(target | (value & mask) << position);
212  return static_cast<T>(target);

393 constexpr T StreamExtract(const uint8_t * stream,
394  size_t size,
395  Mask mask,
396  sjsu::Endian endian = sjsu::Endian::kLittle)
397 {
398  static_assert(
399  std::is_same<T, uint8_t>::value || std::is_same<T, uint16_t>::value ||
400  std::is_same<T, uint32_t>::value || std::is_same<T, uint64_t>::value,
401  "only unsigned integers are allowed to be used in "
402  "sjsu::bit::StreamExtract().");
403 
404  // Diving the starting position by 8 will give you the starting byte position.
405  // The fractional part is truncated, which gives the correct answer. If it
406  // were rounded, the answer would be incorrect and would require the use of
407  // the floor() function.
408  const size_t kStartByte = mask.position / 8;
409  // We need to figure out if the bit position is not byte aligned. We can do
410  // this by taking the modulus of the mask's bit starting position.
411  const size_t kStartingBitOffset = mask.position % 8;
412  // We need to figure out if the bit position of the last bits not byte
413  // aligned. This is simply to determine if we need to consume an extra byte
414  // from the stream.
415  const size_t kEndingBitOffset = (mask.position + mask.width) % 8;
416 
417  // Contains the total number of bytes that will need to extracted from the
418  // byte stream. The number of bytes can be calculated by taking the bit width
419  // desired and dividing it by the size of a byte (8-bits).
420  size_t total_bytes = mask.width / 8;
421 
422  if (kStartingBitOffset != 0)
423  {
424  total_bytes++;
425  }
426 
427  if (kEndingBitOffset != 0)
428  {
429  total_bytes++;
430  }
431 
432  T result = 0;
433 
434  // A pointer to the starting byte of the sequence to pull bytes from.
435  const uint8_t * next_byte_ptr = &stream[kStartByte];
436  // The direction in which we will pull bytes. This is assumed BIG Endian,
437  // thus, the LSB is the first byte. Going forward will pull out more
438  // significant bytes.
439  int scan_direction = 1;
440 
441  if (endian == sjsu::Endian::kLittle)
442  {
443  // Little Endian has its LSB at the end of the array, thus we will pull
444  // bytes from the end of the byte array.
445  next_byte_ptr = &stream[(size - 1) - kStartByte];
446  // To get the more significate bytes, we need to progress through the array
447  // backwards, so we need to increase backwards.
448  scan_direction = -1;
449  }
450 
451  for (uint32_t i = 0; i < total_bytes && kStartByte + i < size; i++)
452  {
453  const int kShiftAmount = (8 * i) - kStartingBitOffset;
454  const uint8_t kNextByte = *next_byte_ptr;
455  next_byte_ptr += scan_direction;
456  T finished_byte = 0;
457 
458  if (kShiftAmount > 0)
459  {
460  finished_byte = kNextByte << kShiftAmount;
461  }
462  else
463  {
464  finished_byte = kNextByte >> -kShiftAmount;
465  }
466  result |= finished_byte;
467  }
468 
469  Mask trimming_mask = mask;
470  trimming_mask.position = 0;
471  result = Extract(result, trimming_mask);
472 
473  return result;
474 }
