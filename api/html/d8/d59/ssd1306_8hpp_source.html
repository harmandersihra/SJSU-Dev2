165  void DrawPixel(int32_t x, int32_t y, Color_t color) override
166  {
167  // The 3 least significant bits hold the bit position within the byte
168  uint32_t bit_position = y & 0b111;
169 
170  // Each byte makes up a vertical column.
171  // Shifting by 3, which also divides by 8 (the 8-bits of a column), will
172  // be the row that we need to edit.
173  uint32_t row = y >> 3;
174 
175  // Mask to clear the bit
176  uint32_t clear_mask = ~(1 << bit_position);
177 
178  // Mask to set the bit, if color.alpha != 0
179  bool pixel_is_on = !color.IsBlank();
180  uint32_t set_mask = pixel_is_on << bit_position;
181 
182  // Address of the pixel column to edit
183  uint8_t * pixel_column = &(bitmap_[row][x]);
184 
185  // Read pixel column and update the pixel
186  uint32_t result = (*pixel_column & clear_mask) | set_mask;
187 
188  // Update pixel with the result of this operation
189  *pixel_column = static_cast<uint8_t>(result);
190  }
