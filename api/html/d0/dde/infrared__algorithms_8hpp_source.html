54 inline bool IsDurationWithinTolerance(
55  uint16_t duration,
56  std::chrono::microseconds expected_duration,
57  float tolerance)
58 {
59  // Calculate the acceptable minimum and maximum ranges:
60  // min = expected_duration * (100% - tolerance)
61  // max = expected_duration * (100% + tolerance)
62  uint16_t min = static_cast<uint16_t>(
63  static_cast<float>(expected_duration.count()) * (1.0f - tolerance));
64  uint16_t max = static_cast<uint16_t>(
65  static_cast<float>(expected_duration.count()) * (1.0f + tolerance));
66  return (min <= duration) && (duration <= max);
67 }

137 inline const DecodedFrame_t Decode(
138  const DataFrame_t * data_frame,
139  const PulseDurationConfiguration_t & configurations)
140 {
141  static constexpr DecodedFrame_t kInvalidFrame = {
142  .data = 0,
143  .is_valid = false,
144  .is_repeat = false,
145  };
146  // Minimum number of pulses to prevent array out of bounds.
147  // Additionally, if the encoding protocol uses repeats sequences, there needs
148  // to be 3 pulses which consists of a header mark, header space, and a stop
149  // mark.
150  constexpr uint8_t kMinimumPulseCount = 3;
151  if (data_frame->pulse_buffer_length < kMinimumPulseCount)
152  {
153  sjsu::LogDebug("Invalid frame too short");
154  return kInvalidFrame;
155  }
156  // Validate the frame's header by checking the first two pulses
157  const uint16_t kHeaderMark = data_frame->pulse_buffer[0];
158  const uint16_t kHeaderSpace = data_frame->pulse_buffer[1];
159  sjsu::LogDebug("%d %d", kHeaderMark, kHeaderSpace);
160  const bool kHeaderMarkIsValid =
161  IsDurationWithinTolerance(kHeaderMark,
162  configurations.header_mark_duration,
163  configurations.tolerance);
164  if (!kHeaderMarkIsValid)
165  {
166  sjsu::LogDebug("Invalid header mark: %d", kHeaderMark);
167  return kInvalidFrame;
168  }
169  // If the encoding utilizes a repeat frame, check for the repeat duration in
170  // the header space before verifying it with the default
171  // header_space_duration. This is because the header space is different in a
172  // repeat sequence.
173  if ((data_frame->pulse_buffer_length == kMinimumPulseCount) &&
174  configurations.uses_repeat_frames)
175  {
176  const uint16_t kStopMark = data_frame->pulse_buffer[2];
177  const bool kRepeatSpaceValid =
178  IsDurationWithinTolerance(kHeaderSpace,
179  configurations.header_repeat_space,
180  configurations.tolerance);
181  const bool kStopMarkValid = IsDurationWithinTolerance(
182  kStopMark, configurations.data_duration, configurations.tolerance);
183  if (kRepeatSpaceValid && kStopMarkValid)
184  {
185  return DecodedFrame_t({
186  .data = 0,
187  .is_valid = true,
188  .is_repeat = true,
189  });
190  }
191  }
192  const bool kHeaderSpaceIsValid =
193  IsDurationWithinTolerance(kHeaderSpace,
194  configurations.header_space_duration,
195  configurations.tolerance);
196  if (!kHeaderSpaceIsValid)
197  {
198  sjsu::LogDebug("Invalid header space: %d", kHeaderSpace);
199  return kInvalidFrame;
200  }
201  // Set all bits to be logically low and only set to high if needed during
202  // decoding.
203  uint32_t decoded_data = 0;
204  // Start decoding the data pulses from the first data pulse edge and stop at
205  // the last post (the stop pulse)
206  //
207  // The following is performed based on the modulation type:
208  // Pulse Distance Modulation:
209  // 1. Verify the fixed mark duration.
210  // 2. Verify the space for the logic level and validity.
211  // Pulse Length Modulation:
212  // 1. Verify the fixed space duration.
213  // 2. Verify the mark for the logic level and validity.
214  for (size_t i = 2; i < data_frame->pulse_buffer_length - 1; i += 2)
215  {
216  const uint16_t kDataMark = data_frame->pulse_buffer[i];
217  const uint16_t kDataSpace = data_frame->pulse_buffer[i + 1];
218  sjsu::LogDebug("%d %d", kDataMark, kDataSpace);
219 
220  uint16_t fixed_data_duration;
221  uint16_t duration_for_logic_level;
222  switch (configurations.encoding_type)
223  {
224  case PulseDurationType::kDistance:
225  fixed_data_duration = kDataMark;
226  duration_for_logic_level = kDataSpace;
227  break;
228  case PulseDurationType::kLength:
229  fixed_data_duration = kDataSpace;
230  duration_for_logic_level = kDataMark;
231  break;
232  }
233 
234  if (!IsDurationWithinTolerance(fixed_data_duration,
235  configurations.data_duration,
236  configurations.tolerance))
237  {
238  return kInvalidFrame;
239  }
240 
241  decoded_data <<= 1;
242  if (IsDurationWithinTolerance(duration_for_logic_level,
243  configurations.logic_high_duration,
244  configurations.tolerance))
245  {
246  decoded_data = sjsu::bit::Set(decoded_data, 0);
247  }
248  else if (!IsDurationWithinTolerance(duration_for_logic_level,
249  configurations.logic_low_duration,
250  configurations.tolerance))
251  {
252  return kInvalidFrame;
253  }
254  // else do nothing, this pulse is valid but the bit is a logical low
255  }
256  return DecodedFrame_t({
257  .data = decoded_data,
258  .is_valid = true,
259  .is_repeat = false,
260  });
261 }
