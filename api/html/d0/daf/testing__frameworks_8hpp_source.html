391 inline void PollingVerification(
392  PollVerificationParameter_t poll,
393  const std::experimental::source_location & location =
394  std::experimental::source_location::current())
395 {
396  // Setup
397  // Setup: Create a polling variable that will track if the polling function is
398  // still polling or not. This is set to false after the polling
399  // function is finished. This variable only become false the polling
400  // function was never locked in a loop or if environmental change due
401  // to the release function results in the polling_function finishing.
402  bool polling = true;
403 
404  // Setup: First run the function that will setup the environment to cause the
405  // polling function to poll/loop/lock.
406  poll.locking_function();
407 
408  std::string file_location =
409  location.file_name() + std::string(":") + std::to_string(location.line());
410 
411  std::string lock_failure_string = file_location;
412  lock_failure_string +=
413  ": locking_function() did not result in the polling_function() locking. "
414  "Failing this test!";
415 
416  std::string release_failure_string = file_location;
417  release_failure_string +=
418  ": polling_function() was not released by release_function(). Failing "
419  "this test!";
420 
421  // Exercise
422  // Exercise: Create the additional thread that will release the
423  // polling_function() from polling after delay_time has elapsed.
424  // Will also verify that that polling variable is correct.
425  std::thread polling_release_thread(
426  [&poll, &polling, &lock_failure_string, &release_failure_string]() {
427  // Wait for polling function to begin polling
428  std::this_thread::sleep_for(poll.delay_time);
429 
430  // Check that polling variable is still TRUE (will become false if
431  // polling finished early).
432  REQUIRE_MESSAGE(polling, lock_failure_string.c_str());
433 
434  // Attempt to release the polling function from polling.
435  poll.release_function();
436 
437  // Wait for polling function to finish.
438  std::this_thread::sleep_for(poll.delay_time);
439 
440  // Check that polling variable is now FALSE
441  REQUIRE_MESSAGE(!polling, release_failure_string.c_str());
442  });
443 
444  // Exercise: Begin polling
445  poll.polling_function();
446 
447  // Polling function has finished at this point, set polling to false.
448  polling = false;
449 
450  // Wait for polling thread to finish.
451  polling_release_thread.join();
452 }
