419  // =========================================================================

425  .Save();

810 inline void SetMaximumClockSpeed()
811 {
812  using sjsu::stm32f10x::SystemController;
813  auto & config =
814  sjsu::SystemController::GetPlatformController()
815  .GetClockConfiguration<SystemController::ClockConfiguration>();
816 
817  // Set the speed of the high speed external oscillator.
818  // NOTE: Change this if its different for your board.
819  config.high_speed_external = 8_MHz;
820  // Set the source of the PLL's oscillator to 4MHz.
821  // See page 93 in RM0008 to see that the internal high speed oscillator,
822  // which is 8MHz, is divided by 2 before being fed to the PLL to get 4MHz.
823  config.pll.source = SystemController::PllSource::kHighSpeedExternal;
824  // Enable PLL to increase the frequency of the system
825  config.pll.enable = true;
826  // Multiply the 8MHz * 9 => 72 MHz
827  config.pll.multiply = SystemController::PllMultiply::kMultiplyBy9;
828  // Set the system clock to the PLL
829  config.system_clock = SystemController::SystemClockSelect::kPll;
830  // APB1's maximum frequency is 36 MHz, so we divide 72 MHz / 2 => 36 MHz.
831  config.ahb.apb1.divider = SystemController::APBDivider::kDivideBy2;
832  // Keep APB1's clock undivided as it can handle up to 72 MHz.
833  config.ahb.apb2.divider = SystemController::APBDivider::kDivideBy1;
834  // Maximum frequency for ADC is 14 MHz, thus the best we can do is divide
835  // 72 MHz / 6 to get 12 MHz.
836  config.ahb.apb2.adc.divider = SystemController::AdcDivider::kDivideBy6;
837  // Set USB to divide by 1.5 which will give you 48 MHz which is needed for USB
838  // usage.
839  config.pll.usb.divider = SystemController::UsbDivider::kDivideBy1Point5;
840 
841  // Initialize platform with new clock configuration settings.
842  sjsu::InitializePlatform();
843 }
