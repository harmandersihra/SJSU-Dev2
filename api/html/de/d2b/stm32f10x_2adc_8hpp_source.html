333  void ModuleInitialize() override
334  {
335  auto & system = sjsu::SystemController::GetPlatformController();
336 
337  // Step 0. Check if the peripheral is already powered on. If it is, then
338  // this peripheral has already been initialized and should not be
339  // initialized again.
340  if (system.IsPeripheralPoweredUp(SystemController::Peripherals::kAdc1))
341  {
342  return;
343  }
344 
345  // Step 1. Power on Peripherals
346  system.PowerUpPeripheral(SystemController::Peripherals::kAdc1);
347  system.PowerUpPeripheral(SystemController::Peripherals::kDma1);
348 
349  // Step 2. Set Power on ADC bit
350  bit::Register(&adc1->CR2).Set(Control2::kAdcOn).Save();
351  sjsu::Delay(1ms);
352 
353  // Step 3. Run Calibration on each peripheral
354  CalibrateADC();
355 
356  // Step 4. Setup DMA for collecting ADC samples
357  SetupDMA();
358 
359  // Step 5. Set ADC control registers and sequence registers
360  adc1->SQR3 = RegularSequence::kRegisterSequence3;
361  adc1->SQR2 = RegularSequence::kRegisterSequence2;
362  adc1->SQR1 = RegularSequence::kRegisterSequence1;
363  adc1->CR1 = Control1::kDefault;
364  adc1->CR2 = Control2::kDefault;
365 
366  // Step 6. Turn on ADC
367  bit::Register(&adc1->CR2).Set(Control2::kAdcOn).Save();
368 
369  // Step 7. Start the conversion process
370  bit::Register(&adc1->CR2).Set(Control2::kSoftwareStart).Save();
371 
372  // Give some time for samples to be collected before leaving initialize
373  Delay(1ms);
