180  void ConfigureFrequency(units::frequency::hertz_t frequency) override
181  {
182  static constexpr auto kUpdateGeneration = bit::MaskFromRange(0);
183 
184  auto & system = SystemController::GetPlatformController();
185  const auto kPeripheralFrequency = system.GetClockRate(pwm_pin_.id);
186 
187  auto period = 1 / frequency;
188  uint32_t product = kPeripheralFrequency * period;
189 
190  pwm_pin_.registers->PSC = GetPrescalarValue(product);
191  pwm_pin_.registers->ARR =
192  static_cast<uint16_t>(product / (pwm_pin_.registers->PSC + 1));
193 
194  bit::Register(&pwm_pin_.registers->EGR).Set(kUpdateGeneration).Save();
195  }

411  uint16_t GetPrescalarValue(uint32_t product)
412  {
413  uint16_t prescalar = 0;
414  uint16_t k_max16_bits = ~0;
415 
416  uint32_t arr;
417 
418  // To select a prescalar and divider value, we use this equation:
419  // (PSC+1)(ARR+1) = (EventTime)(ClkFreq)
420  // product is the value (EventTime)(ClkFreq)
421  //
422  // We want ARR to be a larger value for a more precise duty cycle
423  // so we increment prescalar until ARR in the value (PSC+1)(ARR+1)
424  // is less than 2^16
425  do
426  {
427  prescalar++;
428  arr = product / prescalar;
429  } while (arr > k_max16_bits);
430 
431  return --prescalar;
432  }
