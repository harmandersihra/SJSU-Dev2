238  bool ResponseHandler(const Can::Message_t & message)
239  {
240  if (!(message.length == 8 && message.id == device_id_))
241  {
242  return false;
243  }
244 
245  switch (Commands(message.payload[0]))
246  {
247  case Commands::kReadMotorStatus2:
248  case Commands::kTorqueClosedLoopCommand:
249  case Commands::kSpeedClosedLoopCommand:
250  case Commands::kPositionClosedLoopCommand1:
251  case Commands::kPositionClosedLoopCommand2:
252  case Commands::kPositionClosedLoopCommand3:
253  case Commands::kPositionClosedLoopCommand4:
254  {
255  constexpr float kMax = 33.0f;
256  constexpr int16_t kMaxInt = BitLimits<16, int16_t>::Max();
257 
258  using units::angular_velocity::degrees_per_second_t;
259  using units::current::ampere_t;
260  using units::temperature::celsius_t;
261 
262  auto & payload = message.payload;
263 
264  auto temperature = static_cast<int8_t>(payload[1]);
265  auto current = static_cast<int16_t>((payload[3] << 8) | payload[2]);
266  auto speed = static_cast<int16_t>((payload[5] << 8) | payload[4]);
267  auto encoder = static_cast<int16_t>((payload[7] << 8) | payload[6]);
268 
269  auto temperature_float = static_cast<float>(temperature);
270  auto speed_float = static_cast<float>(speed) / params_.gear_ratio;
271  auto current_float = Map(current, -kMaxInt, kMaxInt, -kMax, kMax);
272 
273  feedback_.encoder_position = encoder;
274  feedback_.temperature = celsius_t{ temperature_float };
275  feedback_.current = ampere_t{ current_float };
276  feedback_.speed = degrees_per_second_t{ speed_float };
277 
278  break;
279  }
280  case Commands::kReadMotorStatus1AndErrorFlagCommands:
281  {
282  using units::temperature::celsius_t;
283  using units::voltage::volt_t;
284 
285  auto & payload = message.payload;
286 
287  uint8_t error = payload[7];
288  auto temperature = static_cast<int8_t>(payload[1]);
289  auto volts = static_cast<uint16_t>((payload[4] << 8) | payload[3]);
290 
291  auto temperature_float = static_cast<float>(temperature);
292  auto volts_float = static_cast<float>(volts);
293

295  feedback_.volts = volt_t{ volts_float * 0.1f };

309  {

337  int32_t ConvertRPMToCommandSpeed(
338  units::angular_velocity::revolutions_per_minute_t rpm,
339  float degree_per_second_ratio = 0.01f)
340  {
341  // Convert RPM to DPS (degrees per second)
342  units::angular_velocity::degrees_per_second_t dps = rpm;
343 
344  // Since the RPM is determined by the input motor speed, in order to make
345  // the output spin at that speed, the DSP must be scaled by the gear_ratio
346  // factor, otherwise for a gear ratio of 2:1, the output speed of the motor
347  // shaft will be half as fast. The RMD motors have a conversion of 0.01
348  // DPS/LSB meaning that the integer value of the scaled DPS must be x100 of
349  // the original value.
350  auto scaled_dps =
351  dps * (1.0f / degree_per_second_ratio) * params_.gear_ratio;
352 
353  // Return the scaled DPS as an int32_t
354  return scaled_dps.to<int32_t>();
355  }
