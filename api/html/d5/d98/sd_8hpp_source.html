625  // Waits for the card to respond after a single or multi block read cmd is
626  // sent.
627  void WaitToReadBlock()
628  {
629  // Since the command encountered no errors, we can now begin to read data.
630  // The card will enter "BUSY" mode following reception of the read command
631  // and sending of a response soon after; we must wait until either the data
632  // token "OxFE" or the error token "b000X_XXXX" is received. The error
633  // token's flags have the following meanings:
634  //
635  // MSB --> 0 (irrelevant)
636  // Bit 6 --> 0 (irrelevant)
637  // Bit 5 --> 0 (irrelevant)
638  // Bit 4 --> If set, card is locked
639  // Bit 3 --> If set, address is out or range
640  // Bit 2 --> If set, card ECC failed
641  // Bit 1 --> If set, CC error occurred
642  // Bit 0 --> If set, a generic error occurred
643 
644  constexpr bit::Mask kErrorIndicator = bit::MaskFromRange(4, 7);
645 
646  for (uint32_t i = 0; i < kRetryLimit * 100; i++)
647  {
648  uint8_t wait_byte = spi_.Transfer(kDontCare);
649 
650  if (wait_byte == 0xFE)
651  {
652  LogDebug("Received GO Byte 0xFE;");
653  LogDebug("Card is now sending block payload...");
654  }
655 
656  if (bit::Extract(wait_byte, kErrorIndicator) == 0x00)
657  {
658  LogDebug("Card Rejected Read Cmd [Response: 0x%02X]", wait_byte);
659  LogDebug(" Error: %s", ToBool(bit::Read(wait_byte, 0)));
660  LogDebug(" CC Error: %s", ToBool(bit::Read(wait_byte, 1)));
661  LogDebug(" Card ECC Failed: %s", ToBool(bit::Read(wait_byte, 2)));
662  LogDebug("Addr Out of Range: %s", ToBool(bit::Read(wait_byte, 3)));
663 
664  throw Exception(std::errc::io_error, "Card Rejected Read Command!");
665  }
666  }

668  throw Exception(std::errc::timed_out, "SD Card did not respond in time.");

763  Response_t SendCommand(Command command,
764  uint32_t parameter,
765  KeepAlive keep_alive)
766  {
767  // Select the SD Card
768  chip_select_.SetLow();
769 
770  // NOTE: So because SanDisk SD Cards are garbage, they need a few clock
771  // cycles as padding before sending the command sequence. We send "0xFF"
772  // because a zero constitutes the start of a command and we don't want to
773  // start the command sequence at this point.
774  spi_.Transfer(kDontCare);
775 
776  // Send command to the SD Card
777  SendCommandParameters(command, parameter);
778 
779  // Creating the response object to return
780  Response_t response;
781 
782  // If the most significant bit of the response byte is a zero, then this
783  // indicates that the start of a response sequence.
784  constexpr bit::Mask kResponseFlag = bit::MaskFromRange(7);
785 
786  for (uint32_t tries = 0; tries < kRetryLimit; tries++)
787  {
789  response.byte[0] = spi_.Transfer(kDontCare);
790 
793  if (bit::Read(response.byte[0], kResponseFlag) == 0)
794  {
795  const uint32_t kResponseLength = GetResponseLength(command);
796 
797  // Store all additional of the response bytes into the response object.
798  for (uint32_t i = 1; i < kResponseLength; i++)
799  {
800  response.byte[i] = spi_.Transfer(kDontCare);
801  }
802  break;
803  }
804  }
805 
806  // Only end the transaction if keep_alive isn't requested
807  if (keep_alive == KeepAlive::kNo)
808  {
809  // Deselect the SD card
810  chip_select_.SetHigh();
811  }
812 
813  return response;
814  }

823  // Initialize and enable the SD Card
824  void Mount()
825  {
826  // As found in the "Application Note Secure Digital Card Interface for the
827  // MSP430" we need to assert the chip select and clock for more than 74
828  // cycles.
829 
830  // So we assert the chip select
831  chip_select_.SetLow();
832 
833  // Clock out 10 bytes or 80 clock cycles, to get the SD Card's internal
834  // state ready for transmission.
835  ClockCard<80>();
836 
837  // Then de-assert chip select
838  chip_select_.SetHigh();
839 
840  // And finally clock for 16 more cycles to finish off the necessary time for
841  // the SD card to finish its internal work.
842  ClockCard<16>();
843 
844  // Reset the card and force it to go to idle state at <400kHz with a
845  // CMD0 + (active-low) CS
846  LogDebug("Resetting SD Card...");
847  SendCommand(Command::kReset, 0, KeepAlive::kNo);
848 
849  // Ask if supported voltage of 3v3 is supported
850  CheckSupportedVoltages();
851 
852  bool initialization_successful = false;
853  for (uint32_t i = 0; i < kRetryLimit * 10; i++)
854  {
855  Response_t begin_response =
856  SendCommand(Command::kAcBegin, 0, KeepAlive::kNo);
857 
858  if (bit::Read(begin_response.byte[0], kIllegalCommand))
859  {
860  throw Exception(std::errc::not_supported,
861  "Card rejected ACMD, this may not be an SD card");
862  }
863 
864  uint32_t init_settings = (0b0101'0000 << 24);
865 
866  Response_t init_response =
867  SendCommand(Command::kAcInit, init_settings, KeepAlive::kNo);
868  // If this bit is a 1, then the init process is still going.
869  // If it is 0, then the init process has finished.
870  if (!bit::Read(init_response.byte[0], kIdle))
871  {
872  LogDebug("Initialization Successful (0x%02X)", init_response.byte[0]);
873  initialization_successful = true;
874  break;
875  }
876  }
877 
878  if (!initialization_successful)
879  {
880  throw Exception(std::errc::timed_out, "Initialization Failed!");
881  }
882 
883  // =========================================================================
884  // Get Card Capacity Type
885  // =========================================================================
886  sd_.type = GetCardType();
887  if (sd_.type == Type::kSDSC)
888  {
889  throw Exception(std::errc::not_supported,
890  "SD Card is standard size. This driver does not support "
891  "Standard Size SD cards.");
892  }
893 
894  LogDebug("SD Card is HC/XC");
895 
896  // =========================================================================
897  // Get and store CSD Register
898  // =========================================================================
899  LogDebug("Getting CSD register contents");
900  sd_.csd = GetCsdRegisterBlock();
901 
902  // Set SPI clock rate to the operating speed set at construction
903  // SPI must be disabled before running configuration methods
904  spi_.Enable(false);
905  spi_.ConfigureFrequency(spi_clock_rate_);
906  spi_.Enable(true);
907  }

910  void CheckSupportedVoltages()
911  {
912  LogDebug("Checking Current SD Card Voltage Level...");
913  constexpr bit::Mask kVoltageCodeMask = bit::MaskFromRange(8, 11) >> 8;
914  constexpr uint8_t kCheckPattern = 0xAB;
915  uint8_t voltage_code = 0x01;
916  uint32_t voltage_pattern = (voltage_code << 8) | kCheckPattern;
917 
918  Response_t response =
919  SendCommand(Command::kGetOp, voltage_pattern, KeepAlive::kYes);
920 
921  LogDebug("Detecting if device is SD version +2.0 or not");
922  if (bit::Read(response.byte[0], kIllegalCommand))
923  {
924  throw Exception(std::errc::not_supported,
925  "Unsupported SD Card responded with Illegal Command");
926  }
927 
928  if (bit::Extract(response.byte[3], kVoltageCodeMask) != voltage_code)
929  {
930  // If the 2nd-to-last byte of the reponse AND with our host device's
931  // supported voltage range is 0x00, the SD card doesn't support our
932  // device's operating voltage
933  LogDebug("Response.byte[3] = 0x%02X", response.byte[3]);
934 
935  throw Exception(std::errc::not_supported,
936  "Unsupported voltage in use. Aborting!");
937  }
938 
939  if (response.byte[4] != kCheckPattern)
940  {
941  // If the last byte is not an exact echo of the LSB of the kGetOp
942  // command's argument, this response is invalid
943  LogDebug("Response.byte[4] = (0x%02X)", response.byte[4]);
944 
945  throw Exception(std::errc::io_error,
946  "Response integrity check failed. Aborting!");
947  }
948  }

983  {
